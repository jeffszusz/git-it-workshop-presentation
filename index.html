Slightly Advanced Tips & Tricks
edit your ~/.gitconfig file: 

[user]
    email = jeff@hackf.org
    name = Jeff Szusz
[core]
    editor = vim
[push]
    default = current
[alias]
    wipe = "reset --hard HEAD"
    standup = "!git log --since yesterday.0:00am --author
      `git config user.email` --pretty=short"
    lds = log --pretty=format:"%C(yellow)%h\\ %ad%Cred%d\\
      %Creset%s%Cblue\\ [%cn\\ %cr]" --decorate --date=short
